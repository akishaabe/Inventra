Daily Sales Summary:
SELECT 
    DATE(s.sale_date) AS sale_day,
    SUM(si.quantity) AS total_items_sold,
    SUM(si.line_total) AS total_revenue
FROM sales s
JOIN sales_items si ON s.sale_id = si.sale_id
GROUP BY sale_day
ORDER BY sale_day;



Product-Level Daily Sales Summary:
SELECT 
    DATE(s.sale_date) AS sale_day,
    p.product_name,
    SUM(si.quantity) AS total_quantity_sold,
    SUM(si.line_total) AS total_sales
FROM sales s
JOIN sales_items si ON s.sale_id = si.sale_id
JOIN products p ON si.product_id = p.product_id
GROUP BY sale_day, p.product_name
ORDER BY sale_day, p.product_name;



Weekly Sales Trends:
SELECT 
    WEEK(s.sale_date, 1) AS week_number,
    p.product_name,
    SUM(si.quantity) AS total_quantity_sold
FROM sales s
JOIN sales_items si ON s.sale_id = si.sale_id
JOIN products p ON si.product_id = p.product_id
GROUP BY week_number, p.product_name
ORDER BY week_number;





Monthly Sales Trends:
SELECT 
    DATE_FORMAT(s.sale_date, '%Y-%m') AS month,
    SUM(si.quantity) AS total_items_sold,
    SUM(si.line_total) AS total_revenue
FROM sales s
JOIN sales_items si ON s.sale_id = si.sale_id
GROUP BY month
ORDER BY month;


Average Unit Per Price:
SELECT 
    p.product_name,
    AVG(si.unit_price) AS avg_unit_price,
    SUM(si.quantity) AS total_quantity_sold
FROM sales_items si
JOIN products p ON si.product_id = p.product_id
GROUP BY p.product_name
ORDER BY avg_unit_price DESC;
